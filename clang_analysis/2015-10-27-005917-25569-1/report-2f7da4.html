364static void process_monitor(unsigned wait_threshold, pid_t watched_process_id)
365{
366
367 sleep(wait_threshold);
368 /*
369 *Even though the checking is performed on errno after the
370 *process is killed, we still have a potential problem
371 *of killing some other processes end up with the
372 *pid same as the one before;
373 *The recycling nature of pid is IGNORED.
374 */
375 errno(*__errno_location ()) = 0;
376 /*
377 *This is a guaranteed kill if the listed 2 cases
378 *do not apply, so SIGKILL is used rather than SIGTERM
379 */
380 if (0 != kill(watched_process_id, SIGKILL9)) {
381 switch (errno(*__errno_location ())) {
382 case EPERM1:
383 /*
384 *This process does not have the permission
385 *to kill the watched process.
386 */
387 case ESRCH3:
388 /*
389 *If the watched process no longer exist,
390 *the monitor action is considered "failed".
391 */
392 pid_array_destroy();
393 exit(EXIT_FAILURE1);
394 break;
395 }
396 }
397 pid_array_destroy();
398 exit(EXIT_SUCCESS0);
399}
